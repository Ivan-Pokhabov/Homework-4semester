\section*{Grammar}

\noindent<Term> ::= <Term'> | <Application> \\
<Application> ::= <Term'> <Spaces> <Term> \\
<Term'> ::= <Abstraction> | <Var> | '(' <Term> ')' \\
<Abstraction> ::= $\lambda$ <VarList> '.' <Term> \\
<VarList> ::= <Var> | <Var> <Spaces> <VarList> \\
<Var> ::= a-z,\{a-z | A-Z0-9 | \_\} \\
<Spaces> ::= '\space' | '\space' <Spaces>
